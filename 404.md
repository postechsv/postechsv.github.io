<script>
  (function() {
    var path = window.location.pathname;
    var decodedPath = decodeURIComponent(path);
    
    // Pattern: /~kmbae or /~kmbae/sub-path
    if (decodedPath.indexOf('/~kmbae') === 0) {
      document.getElementById('redirect-msg').innerHTML = 'Redirecting to new location...';
      
      // Replace /~kmbae (or encoded /%7Ekmbae) with /kmbae
      var newPath = path.replace(/^\/(~|%7[eE])kmbae/, '/kmbae');
      
      // Redirect preserving sub-path, query string, and hash
      window.location.replace(newPath + window.location.search + window.location.hash);
    }
  })();
</script>
